    // 初始化日期
    window.onload = function() {
      const now = new Date();
      now.setMinutes(now.getMinutes() - now.getTimezoneOffset());
      document.getElementById('createDate').value = now.toISOString().slice(0, 16);

    // 動態增減人員
    function addPersonRow() {
      const wrap = document.getElementById("personList");
      const row = document.createElement("div");
      row.className = "person-row";
      row.innerHTML = `
        <input type="text" placeholder="員工代號" />
        <input type="text" placeholder="姓名" />
        <button type="button" class="mini-btn btn-remove" onclick="removePersonRow(this)">－</button>
      `;
      wrap.appendChild(row);
    }

    function removePersonRow(btn) {
      const row = btn.closest(".person-row");
      if (row) row.remove();
    }
