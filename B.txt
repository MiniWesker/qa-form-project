  <script>
    function addNotifyRow() {
      const wrap = document.getElementById("notifyList");
      const row = document.createElement("div");
      row.className = "notify-row";
      row.innerHTML = `
        <label>員工代號：</label>
        <input type="text" placeholder="請輸入代號" />
        <button type="button" class="mini-btn mini-btn-remove" onclick="removeNotifyRow(this)" title="移除人員">-</button>
      `;
      wrap.appendChild(row);
    }

    function removeNotifyRow(btn) {
      const row = btn.closest(".notify-row");
      if (row) {
        // 加入簡單的淡出動畫效果後再移除
        row.style.transition = "opacity 0.2s";
        row.style.opacity = "0";
        setTimeout(() => row.remove(), 200);
      }
    }
  </script>
