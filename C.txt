  <script>
    document.addEventListener("DOMContentLoaded", function() {
      const severityRadios = document.querySelectorAll('input[name="severity"]');
      const frequencyRadios = document.querySelectorAll('input[name="frequency"]');
      const scoreDisplay = document.getElementById('total-score');
      const statusAccept = document.getElementById('status-accept');
      const statusReject = document.getElementById('status-reject');

      function calculateScore() {
        let severity = 0;
        let frequency = 0;
        severityRadios.forEach(radio => { if (radio.checked) severity = parseInt(radio.value); });
        frequencyRadios.forEach(radio => { if (radio.checked) frequency = parseInt(radio.value); });
        const totalScore = severity * frequency;
        scoreDisplay.textContent = totalScore;
        if (severity === 0 || frequency === 0) { scoreDisplay.textContent = "0"; return; }
        if (totalScore > 6) { statusReject.checked = true; statusAccept.checked = false; } 
        else { statusAccept.checked = true; statusReject.checked = false; }
      }

      severityRadios.forEach(radio => radio.addEventListener('change', calculateScore));
      frequencyRadios.forEach(radio => radio.addEventListener('change', calculateScore));
      calculateScore();
      
      const otherInputFields = document.querySelectorAll('.other-input-wrap input[type="text"]');
      otherInputFields.forEach(input => {
          input.addEventListener('input', function() {
              const targetId = this.getAttribute('data-target');
              const targetCheckbox = document.getElementById(targetId);
              if(this.value.trim() !== "") {
                  targetCheckbox.checked = true;
              }
          });
      });
    });
  </script>
